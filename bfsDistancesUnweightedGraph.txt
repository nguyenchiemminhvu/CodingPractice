Given the adjacency matrix of the connected undirected graph with no loops or multiple edges and the index of the start vertex, find the distances between the start vertex and each vertex of the graph.

Example

For

matrix = [[false, false, true],
          [false, false, true],
          [true, true, false]]
and startVertex = 0, the output should be
bfsDistancesUnweightedGraph(matrix, startVertex) = [0, 2, 1].

==============================================================

vector<int> bfsDistancesUnweightedGraph(vector<vector<bool>> matrix, int startVertex)
{
    vector<int> distances (matrix.size(), INT_MAX);
    
    int depth = 0;
    distances[startVertex] = 0;
    
    queue<int> bfs;
    bfs.push(startVertex);
    
    while(!bfs.empty())
    {
        int tam = bfs.size();
        for(int i = 0; i < tam; i++)
        {
            int vertex = bfs.front();
            bfs.pop();
            distances[vertex] = min(distances[vertex], depth);
            for(int j = 0; j < matrix.size(); j++)
            {
                if(matrix[vertex][j] && distances[j] == INT_MAX)
                {
                    bfs.push(j);
                }
            }
        }
        
        depth++;
    }
    
    return distances;
}
