https://leetcode.com/problems/three-divisors/

class Solution 
{
public:
    bool isThree(int n) 
    {
        int count = 0;
        for (int i = 2; i <= n / 2; i++)
        {
            if (n % i == 0)
            {
                count++;
                if (count > 1)
                    return false;
            }
        }
        
        return count == 1;
    }
};
